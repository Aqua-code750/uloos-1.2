    // Shutdown simulator trigger with premium glassmorphic overlay and audio chime
    function shutdownOS() {
      try {
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        const playChimeNote = (freq, delay, duration) => {
          const osc = audioCtx.createOscillator();
          const gain = audioCtx.createGain();
          osc.type = 'triangle';
          osc.frequency.value = freq;
          gain.gain.setValueAtTime(0.15, audioCtx.currentTime + delay);
          gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + delay + duration);
          osc.connect(gain);
          gain.connect(audioCtx.destination);
          osc.start(audioCtx.currentTime + delay);
          osc.stop(audioCtx.currentTime + delay + duration);
        };
        
        playChimeNote(783.99, 0, 0.4);   // G5
        playChimeNote(659.25, 0.15, 0.4); // E5
        playChimeNote(523.25, 0.3, 0.5);  // C5
        playChimeNote(392.00, 0.45, 0.8); // G4
      } catch (e) {
        console.error("Audio Context not supported or allowed yet", e);
      }

      const startMenu = document.getElementById('start-menu');
      if (startMenu) startMenu.classList.remove('active');

      let overlay = document.getElementById('shutdown-overlay');
      if (!overlay) {
        overlay = document.createElement('div');
        overlay.id = 'shutdown-overlay';
        overlay.style.position = 'fixed';
        overlay.style.top = '0';
        overlay.style.left = '0';
        overlay.style.width = '100vw';
        overlay.style.height = '100vh';
        overlay.style.background = 'rgba(15, 23, 42, 0.7)';
        overlay.style.backdropFilter = 'blur(20px)';
        overlay.style.webkitBackdropFilter = 'blur(20px)';
        overlay.style.display = 'flex';
        overlay.style.flexDirection = 'column';
        overlay.style.justifyContent = 'center';
        overlay.style.alignItems = 'center';
        overlay.style.zIndex = '999999';
        overlay.style.color = '#fff';
        overlay.style.fontFamily = "system-ui, -apple-system, sans-serif";
        overlay.style.opacity = '0';
        overlay.style.transition = 'opacity 0.5s ease';
        
        overlay.innerHTML = `
          

            

            
Shutting down UloOS Machine

            
Closing all active virtual environments safely...

          

        `;
        
        if (!document.getElementById('shutdown-spin-style')) {
          const style = document.createElement('style');
          style.id = 'shutdown-spin-style';
          style.innerHTML = `
            @keyframes spin {
              0% { transform: rotate(0deg); }
              100% { transform: rotate(360deg); }
            }
          `;
          document.head.appendChild(style);

        document.body.appendChild(overlay);

      overlay.style.display = 'flex';
      overlay.offsetHeight; // force reflow

        document.querySelectorAll('.window').forEach(win => {

          icon.classList.remove('active');

          overlay.style.display = 'none';
          document.getElementById('boot-screen').style.display = 'flex';
        }, 500);
        
      }, 1800);
    }

    // Dynamic Clock tray updates
    function updateClock() {
      const now = new Date();
      let hours = now.getHours();
      let minutes = now.getMinutes();
      const ampm = hours >= 12 ? 'PM' : 'AM';
      hours = hours % 12;
      hours = hours ? hours : 12;
      minutes = minutes < 10 ? '0' + minutes : minutes;
      document.getElementById('tray-clock').innerText = `${hours}:${minutes} ${ampm}`;
    }

    // Explorer directory populator
    function navigateExplorer(dir) {
      activeExplorerDir = dir;
      document.querySelectorAll('.explorer-folder').forEach(el => el.classList.remove('active'));
      const list = document.querySelectorAll('.explorer-folder');
      if (dir === 'root') list[0].classList.add('active');
      if (dir === 'documents') list[1].classList.add('active');
      if (dir === 'projects') list[2].classList.add('active');
      populateExplorerGrid();
    }

    function populateExplorerGrid() {
      const grid = document.getElementById('explorer-files-grid');
      grid.innerHTML = '';
      const items = vfs[activeExplorerDir];
      items.forEach((item, index) => {
        const div = document.createElement('div');
        div.className = 'explorer-item';
        div.onclick = () => openFileFromExplorer(item, activeExplorerDir);
        
        let icon = 'fa-solid fa-file-code';
        if (item.type === 'folder') icon = 'fa-solid fa-folder';
        else if (item.name.endsWith('.txt')) icon = 'fa-solid fa-file-lines';
        
        div.innerHTML = `
          
          ${item.name}
        `;
        grid.appendChild(div);
      });
    }

    // Explorer Operations
    function triggerFileCreation() {
      const name = prompt("Enter new file name (with extension, e.g. script.js or page.html):");
      if (name) {
        vfs[activeExplorerDir].push({ name: name, type: 'file', content: '// New File' });
        saveVfsToStorage();
        populateExplorerGrid();
        populateIdeSidebar();
      }
    }

    function triggerFolderCreation() {
      const name = prompt("Enter new folder name:");
      if (name) {
        vfs[activeExplorerDir].push({ name: name, type: 'folder', content: '' });
        saveVfsToStorage();
        populateExplorerGrid();
      }
    }

    function triggerFileUpload() {
      const text = prompt("Paste contents of file to import:");
      const name = prompt("Enter file name for import:");
      if (name && text !== null) {
        vfs[activeExplorerDir].push({ name: name, type: 'file', content: text });
        saveVfsToStorage();
        populateExplorerGrid();
        populateIdeSidebar();
      }
    }

    function openFileFromExplorer(item, dir) {
      if (item.name.endsWith('.html') || item.name.endsWith('.js') || item.name.endsWith('.css')) {
        ideActiveFileDir = dir;
        ideActiveFileName = item.name;
        openWindow('win-code');
        loadIdeActiveFile();
      } else {
        openWindow('win-text');
        document.getElementById('text-editor-file-label').innerText = `file: ${item.name}`;
        document.getElementById('ulotext-textarea').value = item.content;
      }
    }

    function toggleIdeFolders() {
      ideFolderCollapsed = !ideFolderCollapsed;
      const fileTree = document.getElementById('ide-file-tree');
      const chevron = document.getElementById('ide-folder-chevron');
      if (ideFolderCollapsed) {

    function openFileInIde(name, dir) {
      // Check if file is already open in tabs
      const alreadyOpen = ideOpenTabs.some(t => t.name === name && t.dir === dir);
      if (!alreadyOpen) {
        ideOpenTabs.push({ name, dir });
      }
      ideActiveFileName = name;
      ideActiveFileDir = dir;
      openWindow('win-code');
      loadIdeActiveFile();
      populateIdeTabs();
      populateIdeSidebar();
      updateGitSidebar();
    }

    function closeIdeTab(name, dir, event) {
      if (event) event.stopPropagation();
      ideOpenTabs = ideOpenTabs.filter(t => !(t.name === name && t.dir === dir));
      
      // If closing active file, pick another
      if (ideActiveFileName === name && ideActiveFileDir === dir) {
        if (ideOpenTabs.length > 0) {
          const nextTab = ideOpenTabs[ideOpenTabs.length - 1];
          ideActiveFileName = nextTab.name;
          ideActiveFileDir = nextTab.dir;
          loadIdeActiveFile();
        } else {
          ideActiveFileName = '';
          ideActiveFileDir = '';
          document.getElementById('ide-code-textarea').value = '';
          document.getElementById('ide-syntax-overlay').innerHTML = '';
          document.getElementById('ide-active-file-label').innerText = 'no active file';
        }
      }
      populateIdeTabs();
      populateIdeSidebar();
    }

    function populateIdeTabs() {
      const container = document.getElementById('ide-tabs-container');
      container.innerHTML = '';
      ideOpenTabs.forEach(tab => {
        const tabDiv = document.createElement('div');
        tabDiv.className = `ide-tab ${tab.name === ideActiveFileName && tab.dir === ideActiveFileDir ? 'active' : ''}`;
        tabDiv.onclick = () => {
          ideActiveFileName = tab.name;
          ideActiveFileDir = tab.dir;
          loadIdeActiveFile();
          populateIdeTabs();
        };

        // File icon based on extension
        let icon = '';
        if (tab.name.endsWith('.html')) icon = '';
        else if (tab.name.endsWith('.css')) icon = '';
        else if (tab.name.endsWith('.js')) icon = '';

        tabDiv.innerHTML = `
          ${icon}
          ${tab.name}
          
        `;
        container.appendChild(tabDiv);
      });
    }

    function populateIdeSidebar() {
      const list = document.getElementById('ide-file-tree');
      list.innerHTML = '';
      const projects = vfs['projects'];
      projects.forEach(file => {
        const item = document.createElement('div');
        item.style.cssText = `padding:6px 8px; font-size:11px; cursor:pointer; border-radius:4px; display:flex; align-items:center; gap:6px; ${ideActiveFileName === file.name ? 'background:rgba(0, 240, 255, 0.1);color:#00f0ff;font-weight:600;' : 'color:#cbd5e1;'}`;
        item.onclick = () => {
          openFileInIde(file.name, 'projects');
        };

        let fileIcon = '';
        if (file.name.endsWith('.html')) fileIcon = '';
        else if (file.name.endsWith('.css')) fileIcon = '';
        else if (file.name.endsWith('.js')) fileIcon = '';

        item.innerHTML = `${fileIcon} ${file.name}`;
        list.appendChild(item);
      });
    }

    function loadIdeActiveFile() {
      if (!ideActiveFileName) return;
      const file = vfs[ideActiveFileDir].find(f => f.name === ideActiveFileName);
      if (file) {
        document.getElementById('ide-active-file-label').innerHTML = ` editing: ${ideActiveFileName}`;
        
        let lang = 'HTML5';
        if (ideActiveFileName.endsWith('.css')) lang = 'CSS3';
        else if (ideActiveFileName.endsWith('.js')) lang = 'Javascript';
        document.getElementById('ide-lang-info').innerText = lang;

        const textarea = document.getElementById('ide-code-textarea');
        textarea.value = file.content;
        
        // Populate tabs
        populateIdeTabs();
        
        // Trigger code highlighting, gutter sync
        handleIdeEditorInput();
        
        // Debounced preview
        triggerDebouncedPreview();
      }
    }

    function saveIdeCode() {
      if (!ideActiveFileName) return;
      const file = vfs[ideActiveFileDir].find(f => f.name === ideActiveFileName);
      if (file) {
        file.content = document.getElementById('ide-code-textarea').value;
        saveVfsToStorage();
        updateGitSidebar();
        // Spawns a floating sleek notification instead of blocking alert!
        showSystemNotification("UloCode Studio", "File successfully synchronized and committed to local VFS.");
      }
    }
